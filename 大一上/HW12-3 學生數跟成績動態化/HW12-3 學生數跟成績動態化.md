# 作業 12-3：學生數跟成績動態化 (Ultimate Dynamic List)

這是大一上 C++ 課程的集大成之作！我們實作了一個完全動態的學生成績管理系統：不僅班級數量由使用者決定，每個班級內的學生人數也能在執行時動態輸入。透過巢狀動態記憶體配置 (Dynamic Class -> Dynamic Student)，程式展現了極高的彈性與專業的指標操作邏輯。

## 💡 程式碼與解題思路

完全動態化的巢狀串列邏輯，已詳細對齊在右側註解區：

```cpp
#include<iostream>
#include<cstdlib>
using namespace std;
struct student {                                            // 定義學生節點
	int id;                                                 // 學號
	float grade;                                            // 成績 (使用浮點數)
	student* next;                                          // 指向下一位學生的指標
};
struct classes {                                            // 定義班級節點
	int students;                                           // 紀錄該班級的學生人數
	student* first_student;                                 // 指向班級第一位學生的指標
	classes* next;                                          // 指向下一班的指標
};
void add_student(classes* current, int id, float grade) {   // 副程式：在指定班級內動態新增學生
	student* new_s = (student*)malloc(sizeof(student));     // 申請新學生空間
	new_s->id = id; new_s->grade = grade; new_s->next = NULL;
	if (current->first_student == NULL) {                   // 如果班級目前沒有人
		current->first_student = new_s;                     // 該生即為首位學生
	}
	else {                                                  // 如果已經有人，則走訪至串列末端
		student* temp = current->first_student;
		while (temp->next != NULL) { temp = temp->next; }    // 找到最後一個節點
		temp->next = new_s;                                 // 將新學生串接在末尾
	}
}
void show(classes* head) {                                  // 副程式：遍歷輸出所有動態班級與學生
	int i = 1;
	for (classes* p = head; p != NULL; p = p->next, i++) {  // 外層走訪班級
		cout << "Class" << i << "(" << p->students << "):";
		for (student* s = p->first_student; s != NULL; s = s->next) { // 內層走訪學生
			cout << s->id << "(" << s->grade << ")->";
		}
		cout << "NULL" << endl;
	}
}
int main(void) {                                            // 進入主程式
	int n; cin >> n;                                        // 讀取班級總數
	classes* head = NULL; classes* current = NULL;
	for (int i = 0; i < n; i++) {                           // 依照輸入次數，動態產生班級節點
		classes* newnode = (classes*)malloc(sizeof(classes));
		newnode->first_student = NULL; newnode->next = NULL;
		if (head == NULL) { head = newnode; current = newnode; } // 處理首個班級
		else { current->next = newnode; current = newnode; }     // 串接後續班級
		int m; cin >> m;                                    // 讀取該班學生人數
		current->students = m;
		for (int j = 0; j < m; j++) {                       // 依照人數動態新增學生
			float grade; cin >> grade;
			add_student(current, j + 1, grade);             // 呼叫副程式完成資料存入
		}
	}
	show(head);                                             // 展現最終動態生成的巢狀串列
	return 0;
}
```
