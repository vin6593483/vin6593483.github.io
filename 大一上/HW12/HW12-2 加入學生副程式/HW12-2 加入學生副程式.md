# 作業 12-2：加入學生副程式 (Dynamic Nesting)

這題將巢狀鏈結串列的操作進一步優化。透過自訂 `add_student` 副程式，我們不需要在主程式中重複撰寫複雜的 `malloc` 與指標串接邏輯。副程式會自動判斷當前班級是否為空，並正確地將新學生節點掛載到串列末尾。這種封裝邏輯的方式，是開發大型系統時非常重要的技巧。

## 💡 程式碼與解題思路

詳細的副程式判斷邏輯與指標偏移細節，我都對齊在右側註解供大家參考：

```cpp
#include<iostream>
#include<cstdlib>
using namespace std;
struct student {                                            // 定義學生節點
	int id;                                                 // 學號
	int grade;                                              // 成績
	student* next_student;                                  // 指向下一位學生的指標
};
struct  classes {                                           // 定義班級節點
	int students;                                           // 紀錄該班級目前的學生總數
	student* first_student;                                 // 指向班級第一位學生的指標
	classes* next_class;                                    // 指向各班級間串接的指標
};
classes* c;                                                 // 全域變數：指向當前操作的班級
student* s;                                                 // 全域變數：指向當前班級的最後一位學生
void add_student(int id, int grade) {                       // 副程式：動態新增學生到目前班級
	if (c->students == 0) {                                 // 如果該班級還沒有學生
		c->first_student = NULL;                            // 先確保首位學生指標為空
	}
	if (c->first_student == NULL) {                         // 處理班級的第一位學生
		c->first_student = (student*)malloc(sizeof(student)); // 申請首位學生的記憶體空間
		s = c->first_student;                               // 將走訪指標 s 移至首位
	}
	else if (s->next_student == NULL) {                     // 處理後續新增的學生
		s->next_student = (student*)malloc(sizeof(student)); // 申請新學生的空間並串接在後
		s = s->next_student;                                // 將指標 s 移向最新的學生節點
	}
	s->id = id;                                             // 設定學號
	s->grade = grade;                                       // 設定成績
	s->next_student = NULL;                                 // 確保新節點的下一節為空
	c->students++;                                          // 班級人數計數加 1
}
void show(classes* class1) {                                // 副程式：遍歷輸出所有班級成績
	int i = 1;
	classes* now_c = class1;                                // 使用區域指標走訪，不破壞全域地址
	student* now_s = s;
	for (now_c = class1; now_c != NULL; now_c = now_c->next_class, i++) {
		cout << "Class" << i << "(" << now_c->students << ")" << ":"; // 輸出班級編號與人數
		for (now_s = now_c->first_student; now_s != NULL; now_s = now_s->next_student) {
			cout << now_s->id << "(" << now_s->grade << ")->";
		}
		cout << "NULL" << endl;
	}
}
int main() {                                                // 進入主程式
	int c1s1, c1s2, c1s3, c2s1, c2s2, c3s1, c3s2, c3s3;     // 宣告變數接收各班成績輸入
	cin >> c1s1 >> c1s2 >> c1s3 >> c2s1 >> c2s2 >> c3s1 >> c3s2 >> c3s3;
	classes* class1 = (classes*)malloc(sizeof(classes));    // 建立一班空間
	class1->students = 0; c = class1;
	add_student(1, c1s1); add_student(2, c1s2); add_student(3, c1s3); // 呼叫副程式新增學生
	c->next_class = (classes*)malloc(sizeof(classes));      // 建立二班空間
	c = c->next_class; c->students = 0;
	add_student(1, c2s1); add_student(2, c2s2);
	c->next_class = (classes*)malloc(sizeof(classes));      // 建立三班空間
	c = c->next_class; c->students = 0;
	add_student(1, c3s1); add_student(2, c3s2); add_student(3, c3s3);
	c->next_class = NULL;                                   // 結束所有班級串聯
	show(class1);                                           // 展現成果
	system("pause");
	return 0;
}
```
