# 作業 11-2：副程式找最小值及平均值

這題進一步練習 Linked List 的副程式應用。我們手動建立了 5 個 node 節點並將其位址串連起來，接著將頭節點的位址分別傳入兩個副程式。一個副程式負責巡視整個串列尋找最小值，另一個則負責加總並計算平均值，展現了 Linked List 在資料處理上的靈活性。

## 💡 程式碼與解題思路

詳細的副程式邏輯與指標移動過程，我都對齊在右側註解供大家參考：

```cpp
#include<iostream>
using namespace std;
struct node {                                               // 定義節點結構
	int data;                                               // 存放資料的變數
	node* next;                                             // 指向下一節 node 結構的指標
};
int min(node* ptr) {                                        // 副程式：尋找最小值，接收頭節點指標
	int mininum = ptr->data;                                // 預設第一個節點的資料為最小值
	while (ptr != NULL) {                                   // 只要指標還沒指到空位 (NULL)，就繼續走訪
		if (ptr->data < mininum) {                          // 如果當前節點的值比紀錄的最小值還小
			mininum = ptr->data;                            // 更新最小值
		}
		ptr = ptr->next;                                    // 指標移向下一節
	}
	return mininum;                                         // 回傳最終找出的最小值
}
float avg(node* ptr) {                                      // 副程式：計算平均值
	float sum = 0;                                          // 宣告累加變數 sum
	while (ptr != NULL) {                                   // 指標走訪整個串列
		sum = sum + ptr->data;                              // 將每個節點的資料累加
		ptr = ptr->next;                                    // 指標往後移動一節
	}
	float average = (sum / 5);                              // 計算平均值 (總和除以節點總數 5)
	return average;                                         // 回傳計算結果
}
int main(void) {                                            // 進入主程式
	node s1, s2, s3, s4, s5;                                // 手動宣告 5 個 node 變數
	cin >> s1.data >> s2.data >> s3.data >> s4.data >> s5.data; // 讀取 5 筆輸入資料
	s1.next = &s2;                                          // 將 s1 指向 s2 的地址
	s2.next = &s3;                                          // 將 s2 指向 s3
	s3.next = &s4;                                          // 將 s3 指向 s4
	s4.next = &s5;                                          // 將 s4 指向 s5
	s5.next = NULL;                                         // 最後一節 s5 指向 NULL 代表結束
	cout << min(&s1) << " ";                                // 呼叫 min 副程式並輸出結果
	cout << avg(&s1) << endl;                               // 呼叫 avg 副程式並輸出結果
	return 0;                                               // 程式順利結束
}
```
