# 作業 11-3：插入與刪除節點

這題是 Linked List 的進階實作，涵蓋了「搜尋 (FIND)」、「插入 (ADD)」與「刪除 (DEL)」三大核心功能。程式的核心在於指標的重新導向：插入時需將新節點接入鏈結中，刪除時則需跳過目標節點將前後串接。此外，程式也實作了基本的防呆機制，確保指令與資料存在才進行操作。

## 💡 程式碼與解題思路

詳細的指標操作與鏈結重組邏輯，我都對齊在右側註解供大家參考：

```cpp
#include<iostream>
using namespace std;
struct node {                                               // 定義節點結構
	int num;                                                // 存放資料
	node* next_num;                                         // 指向下一節的指標
};
int FIND(int n,node*ptr) {                                  // 副程式：尋找特定數值是否存在
	int goal = 0;                                           // 找到的次數計數器
	while (ptr != NULL) {                                   // 遍歷串列
		if (ptr->num == n) { goal++; }                      // 找到目標則計數
		ptr = ptr->next_num;                                // 指標往後移
	}
	return goal;                                            // 回傳結果
}
void ADD(node*first, int n,node*n_add) {                    // 副程式：在數字 n 之後插入新節點 n_add
	node* ptr = first;                                      // 從頭開始尋找插入點
	while (ptr != NULL && ptr->num != n) { ptr = ptr->next_num; }
	if (ptr != NULL && ptr->num == n) {                     // 找到插入位置後
		n_add->next_num = ptr->next_num;                    // 新節點指向原本的下一節
		ptr->next_num = n_add;                              // 原本位置指向新節點，完成插入
		for (node* p = first; p != NULL; p = p->next_num) { cout << p->num << "->"; }
		cout << "NULL";                                     // 輸出插入後的結果
	}
}
void DEL(node* first, int d) {                              // 副程式：刪除數值為 d 的節點
	if (first->num == d) {                                  // 特殊狀況：如果要刪除的是第一個節點
		for (node* p = first->next_num; p != NULL; p = p->next_num) { cout << p->num << "->"; }
		cout << "NULL"; return;
	}
	node* ptr = first; node* last = NULL;                   // 使用 last 指標紀錄前一個節點
	while (ptr != NULL && ptr->num != d) { last = ptr; ptr = ptr->next_num; }
	if (ptr != NULL && ptr->num == d) {                     // 找到目標節點後
		last->next_num = ptr->next_num;                     // 前一個節點跳過目標，直接指向下一節，完成刪除
		for (node* p = first; p != NULL; p = p->next_num) { cout << p->num << "->"; }
		cout << "NULL";
	}
}
int main(void) {                                            // 進入主程式
	node n1, n2, n3, n4, n5; node n_add;                    // 宣告節點空間
	n1.num = 1; n2.num = 2; n3.num = 3; n4.num = 4; n5.num = 5; // 初始化數值
	n1.next_num = &n2; n2.next_num = &n3; n3.next_num = &n4; n4.next_num = &n5; n5.next_num = NULL;
	for (node* p = &n1; p != NULL; p = p->next_num) { cout << p->num << "->"; } // 輸出原始串列
	cout << "NULL" << endl;
	char input; cin >> input;                               // 讀取指令 (a:插入, d:刪除)
	if (input == 'a') {                                     // 插入邏輯處理
		int n, m; cin >> n >> m;
		if (FIND(n, &n1) == 0) { cout << "查無此數字"; return 0; }
		n_add.num = m; ADD(&n1, n, &n_add);
	} else if (input == 'd') {                              // 刪除邏輯處理
		int d; cin >> d;
		if (FIND(d, &n1) == 0) { cout << "查無此數字"; return 0; }
		DEL(&n1, d);
	} else { cout << "無此指令"; }                            // 錯誤指令處理
	return 0;
}
```
